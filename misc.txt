# helper to parse date from form (YYYY-MM-DD)
    def parse_date(s):
        try:
            return datetime.strptime(s, "%Y-%m-%d").date()
        except (TypeError, ValueError):
            return None

    rateapplied = request.form.get('rateapplied') or ''

    origdp_str     = request.form.get('origdp') or ''
    refixeddp_str  = request.form.get('refixeddp') or ''
    extendeddp_str = request.form.get('extendeddp') or ''
    supply_str     = request.form.get('supdate') or ''

    origdp     = parse_date(origdp_str)
    refixeddp  = parse_date(refixeddp_str)
    extendeddp = parse_date(extendeddp_str)
    supply     = parse_date(supply_str)

    # --------- VALIDATION LOGIC ---------
    if supply and (origdp or refixeddp or extendeddp):

        # compute last allowed "due date" among present DP dates
        due_dates = [d for d in [origdp, refixeddp, extendeddp] if d]
        last_due = max(due_dates) if due_dates else None

        # Case: supply before or on last due date → only "Supply before due date" valid
        if last_due and supply <= last_due:
            if rateapplied != "Supply before due date":
                flash("For supply on/before due date, only 'Supply before due date' is allowed.", "danger")
                return redirect(url_for('index'))

        # Case: supply after last due date → 'Supply after due date' or 'Lower rate Applicable' 
        if supply > last_due:
            if rateapplied not in [
                "Supply after due date",
                "Lower rate Applicable",
                "Lower rate and LD comparison",
                "Lower rate with LD in further extension",
            ]:
                flash("For delayed supply, choose a valid option.", "danger")
                return redirect(url_for('index'))
